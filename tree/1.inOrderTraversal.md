94. Binary Tree Inorder Traversal

// Basic: Recursive inorder traversal
// tc: O(n), sc: O(h) where h is the height of the tree (O(log n) for balanced, O(n) for skewed)

```cpp
class Solution {
private:
    void traverse(TreeNode* root, vector<int>& res) {
        if (root == nullptr)
            return;
        traverse(root->left, res);
        res.push_back(root->val);
        traverse(root->right, res);
    }

public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        traverse(root, res);
        return res;
    }
};
```

// Better: Iterative inorder traversal using stack
// tc: O(n), sc: O(h)

```cpp
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> st;
        TreeNode* curr = root;
        while (curr != nullptr || !st.empty()) {
            while (curr != nullptr) {
                st.push(curr);
                curr = curr->left;
            }
            curr = st.top(); st.pop();
            res.push_back(curr->val);
            curr = curr->right;
        }
        return res;
    }
};
```

// Best: Morris inorder traversal (O(1) space)
// tc: O(n), sc: O(1)

```cpp
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        TreeNode* curr = root;
        while (curr != nullptr) {
            if (curr->left == nullptr) {
                res.push_back(curr->val);
                curr = curr->right;
            } else {
                TreeNode* pre = curr->left;
                while (pre->right != nullptr && pre->right != curr) {
                    pre = pre->right;
                }
                if (pre->right == nullptr) {
                    pre->right = curr;
                    curr = curr->left;
                } else {
                    pre->right = nullptr;
                    res.push_back(curr->val);
                    curr = curr->right;
                }
            }
        }
        return res;
    }
};
```
